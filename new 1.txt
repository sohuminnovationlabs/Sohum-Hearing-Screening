// GPS Location with Smart Caching
let lastLocation = null;
let lastLocationTime = null;

const getSmartLocation = () => {
  return new Promise((resolve) => {
    const now = Date.now();
    const CACHE_DURATION = 15 * 60 * 1000; // 15 minutes
    
    // Use cached location if recent
    if (lastLocation && lastLocationTime && (now - lastLocationTime) < CACHE_DURATION) {
      resolve(lastLocation);
      return;
    }
    
    // Try to get new location
    if (!navigator.geolocation) {
      resolve({ latitude: '', longitude: '', accuracy: 'Not available' });
      return;
    }
    
    navigator.geolocation.getCurrentPosition(
      (position) => {
        lastLocation = {
          latitude: position.coords.latitude.toFixed(6),
          longitude: position.coords.longitude.toFixed(6),
          accuracy: position.coords.accuracy ? `${Math.round(position.coords.accuracy)}m` : 'Unknown'
        };
        lastLocationTime = now;
        resolve(lastLocation);
      },
      (error) => {
        // If error, use last known location or empty
        resolve(lastLocation || { latitude: '', longitude: '', accuracy: 'Permission denied' });
      },
      {
        timeout: 5000,
        maximumAge: 60000, // Accept 1-min-old location from device
        enableHighAccuracy: false // Use network location (WiFi/cell), not GPS
      }
    );
  });
};

// Enhanced Stats Calculation
const calculateDetailedStats = (logData, operatorId) => {
  const dataRows = logData.slice(1); // Skip header row
  const userScreenings = dataRows.filter(row => row[2] === operatorId);
  if (userScreenings.length === 0) {
    return { today: 0, thisWeek: 0, thisMonth: 0, last30Days: 0, allTime: 0, dailyAvg: 0, bestDay: null, aabrCount: 0, oaeCount: 0 };
  }
  const now = new Date();
  const todayStart = new Date(now.getFullYear(), now.getMonth(), now.getDate());
  const weekStart = new Date(now.getTime() - 7 * 24 * 60 * 60 * 1000);
  const monthStart = new Date(now.getFullYear(), now.getMonth(), 1);
  const thirtyDaysAgo = new Date(now.getTime() - 30 * 24 * 60 * 60 * 1000);
  const today = userScreenings.filter(row => new Date(row[0]) >= todayStart).length;
  const thisWeek = userScreenings.filter(row => new Date(row[0]) >= weekStart).length;
  const thisMonth = userScreenings.filter(row => new Date(row[0]) >= monthStart).length;
  const last30Days = userScreenings.filter(row => new Date(row[0]) >= thirtyDaysAgo).length;
  const allTime = userScreenings.length;
  const daysIntoMonth = now.getDate();
  const dailyAvg = daysIntoMonth > 0 ? (thisMonth / daysIntoMonth).toFixed(1) : 0;
  const dayGroups = {};
  userScreenings.forEach(row => {
    const dateStr = new Date(row[0]).toISOString().split('T')[0];
    dayGroups[dateStr] = (dayGroups[dateStr] || 0) + 1;
  });
  const bestDayEntry = Object.entries(dayGroups).sort((a, b) => b[1] - a[1])[0];
  const bestDay = bestDayEntry ? { count: bestDayEntry[1], date: new Date(bestDayEntry[0]).toLocaleDateString('en-US', { month: 'short', day: 'numeric' }) } : null;
  const thisMonthScreenings = userScreenings.filter(row => new Date(row[0]) >= monthStart);
  const aabrCount = thisMonthScreenings.filter(row => row[17] === 'AABR').length;
  const oaeCount = thisMonthScreenings.filter(row => row[17] === 'OAE').length;
  return { today, thisWeek, thisMonth, last30Days, allTime, dailyAvg: parseFloat(dailyAvg), bestDay, aabrCount, oaeCount };
};

// SNR Stepper Component with Scroll Wheel
const SNRStepper = ({ value, onChange, label }) => {
  const [showWheel, setShowWheel] = useState(false);
  const [wheelValue, setWheelValue] = useState(value || 0.0);
  
  // Generate values from 0.0 to 30.0 in 0.1 increments, plus ">30.0"
  const allValues = [...Array(301).keys()].map(i => (i * 0.1).toFixed(1));
  allValues.push('>30.0');
  
  const currentValue = value || 0.0;
  
  const adjustValue = (delta) => {
    const current = parseFloat(currentValue);
    if (isNaN(current)) return;
    let newVal = current + delta;
    if (newVal < 0) newVal = 0;
    if (newVal > 30) newVal = 30;
    onChange(newVal.toFixed(1));
  };
  
  const openWheel = () => {
    setWheelValue(currentValue);
    setShowWheel(true);
  };
  
  const saveWheel = () => {
    onChange(wheelValue);
    setShowWheel(false);
  };
  
return React.createElement('div', { className: 'space-y-1' },
  React.createElement('label', { className: 'text-xs font-medium text-gray-600 ml-1 block' }, label),
  React.createElement('div', { className: 'flex items-center gap-1' },
    React.createElement('button', {
      type: 'button',
      onClick: () => adjustValue(-1.0),
      className: 'w-10 h-10 bg-gray-200 rounded-lg font-bold text-sm text-gray-700 hover:bg-gray-300 active:bg-gray-400 transition-colors'
    }, '−−'),
    React.createElement('button', {
      type: 'button',
      onClick: () => adjustValue(-0.1),
      className: 'w-10 h-10 bg-gray-200 rounded-lg font-bold text-sm text-gray-700 hover:bg-gray-300 active:bg-gray-400 transition-colors'
    }, '−'),
    React.createElement('button', {
      type: 'button',
      onClick: openWheel,
      className: 'flex-1 h-10 bg-primary text-white rounded-lg font-bold text-base hover:bg-primary/90 transition-colors shadow-sm'
    }, `${currentValue} dB`),
    React.createElement('button', {
      type: 'button',
      onClick: () => adjustValue(0.1),
      className: 'w-10 h-10 bg-gray-200 rounded-lg font-bold text-sm text-gray-700 hover:bg-gray-300 active:bg-gray-400 transition-colors'
    }, '+'),
    React.createElement('button', {
      type: 'button',
      onClick: () => adjustValue(1.0),
      className: 'w-10 h-10 bg-gray-200 rounded-lg font-bold text-sm text-gray-700 hover:bg-gray-300 active:bg-gray-400 transition-colors'
    }, '++')
  ),
    showWheel && React.createElement('div', {
      className: 'fixed inset-0 bg-black/50 flex items-center justify-center z-50',
      onClick: () => setShowWheel(false)
    },
      React.createElement('div', {
        className: 'bg-white rounded-2xl p-6 w-80 shadow-xl',
        onClick: (e) => e.stopPropagation()
      },
        React.createElement('h3', { className: 'text-lg font-semibold text-gray-800 mb-4 text-center' }, label),
        React.createElement('select', {
          value: wheelValue,
          onChange: (e) => setWheelValue(e.target.value),
          size: 10,
          className: 'w-full text-center text-2xl font-bold border-2 border-primary rounded-xl p-4 focus:outline-none focus:ring-2 focus:ring-primary/50'
        },
          allValues.map(val => 
            React.createElement('option', { key: val, value: val, className: 'text-xl py-2' }, `${val} dB`)
          )
        ),
        React.createElement('div', { className: 'flex gap-3 mt-6' },
          React.createElement('button', {
            onClick: () => setShowWheel(false),
            className: 'flex-1 py-3 px-4 bg-gray-200 text-gray-700 rounded-xl font-medium hover:bg-gray-300 transition-colors'
          }, 'Cancel'),
          React.createElement('button', {
            onClick: saveWheel,
            className: 'flex-1 py-3 px-4 bg-primary text-white rounded-xl font-medium hover:bg-primary/90 transition-colors'
          }, 'Done')
        )
      )
    )
  );
};

const compressPhoto = (file) => {
  return new Promise((resolve, reject) => {
    const reader = new FileReader();
    reader.onerror = () => reject(new Error('File read failed'));
    reader.onload = (e) => {
      const img = new Image();
      img.onerror = () => reject(new Error('Image load failed'));
      img.onload = () => {
        const canvas = document.createElement('canvas');
        let w = img.width;
        let h = img.height;
        if (w > PHOTO_MAX_WIDTH || h > PHOTO_MAX_HEIGHT) {
          const ratio = Math.min(PHOTO_MAX_WIDTH / w, PHOTO_MAX_HEIGHT / h);
          w = Math.round(w * ratio);
          h = Math.round(h * ratio);
        }
        canvas.width = w;
        canvas.height = h;
        const ctx = canvas.getContext('2d');
        ctx.drawImage(img, 0, 0, w, h);
        if (PHOTO_GRAYSCALE) {
          const imageData = ctx.getImageData(0, 0, w, h);
          const pixels = imageData.data;
          for (let i = 0; i < pixels.length; i += 4) {
            const grey = Math.round(0.299 * pixels[i] + 0.587 * pixels[i + 1] + 0.114 * pixels[i + 2]);
            pixels[i] = grey;
            pixels[i + 1] = grey;
            pixels[i + 2] = grey;
          }
          ctx.putImageData(imageData, 0, 0);
        }
        const base64 = canvas.toDataURL('image/jpeg', PHOTO_QUALITY).split(',')[1];
        resolve({ base64, canvas });
      };
      img.src = e.target.result;
    };
    reader.readAsDataURL(file);
  });
};

const isImageBlurry = (canvas) => {
  const ctx = canvas.getContext('2d');
  const { width, height } = canvas;
  const imageData = ctx.getImageData(0, 0, width, height);
  const pixels = imageData.data;
  const greyPixels = [];
  for (let i = 0; i < pixels.length; i += 4) {
    greyPixels.push(0.299 * pixels[i] + 0.587 * pixels[i + 1] + 0.114 * pixels[i + 2]);
  }
  const mean = greyPixels.reduce((a, b) => a + b, 0) / greyPixels.length;
  const variance = greyPixels.reduce((a, b) => a + Math.pow(b - mean, 2), 0) / greyPixels.length;
  return variance < 100;
};

const uploadPhotoToDrive = async (base64Data, screeningId) => {
  const MAX_ATTEMPTS = 3;
  const DELAY_MS = 2000;
  const sleep = (ms) => new Promise(res => setTimeout(res, ms));
  for (let attempt = 1; attempt <= MAX_ATTEMPTS; attempt++) {
    try {
      await fetch(SCRIPT_URL + '?action=uploadPhoto', {
        method: 'POST',
        mode: 'no-cors',
        headers: { 'Content-Type': 'text/plain' },
        body: JSON.stringify({
          base64Data,
          screeningId,
          filename: `${screeningId}.jpg`
        })
      });
      return 'pending';
    } catch (err) {
      console.warn(`Photo upload attempt ${attempt} failed:`, err);
    }
    if (attempt < MAX_ATTEMPTS) await sleep(DELAY_MS);
  }
  return null;
};
